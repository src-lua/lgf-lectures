NODE query(int no, int l, int r, int a, int b) { // SegTree
    if (b < l || r < a) return NODE();
    if (a <= l && r <= b) return seg[no];
    int m = (l + r) >> 1;
    return NODE::merge(query(no << 1, l, m, a, b),
                    query((no << 1) | 1, m + 1, r, a, b));
}
